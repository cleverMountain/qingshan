# Object.create

Object.prototype.create = function (proto, propertiesObject) {
  let obj = {}
  obj.__proto__ = proto
  if (propertiesObject) {
    for (let key in propertiesObject) {
      if (propertiesObject.hasOwnProperty(key)) {
        Object.defineProperty(obj, key, propertiesObject[key])
      }
    }
  }
  return obj
}
